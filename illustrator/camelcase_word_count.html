function camelcase($s)
{
    if (empty($s) || !is_string($s))
        return 0;
    $length = strlen($s);
    if ($length < 1 || $length > 100000)
        return 0;
    preg_match_all('/[A-Z]+/m', $s, $matches, PREG_SET_ORDER, 0);

    $wordCount = count($matches) + 1;
    if (!empty($matches) && strpos($s, $matches[0][0]) == 0)
        return 0;
    return $wordCount;
}
